# nodemailer
nodemailer sample

```
const express = require('express');
const request = require('request');
const fs = require('fs');
const nodemailer = require('nodemailer');

const app = express();

app.get('/', (req, res) => {
   const transporter = nodemailer.createTransport({
      service: 'Naver',
      prot: 587,
      host: 'smtp.naver.com',  
      secure: false,  
      requireTLS: true ,
      auth: {
        user: 'naverEmail',
        pass: 'naverPassword'
      }
    });
  
    const mailOptions = {
      from: 'fromEmail',
      to: 'toEmail',
      subject: 'test',
      text: 'hello'
    };
      
    transporter.sendMail(mailOptions, function (error, info) {
      if (error) {
        console.log(error);
      } else {
        console.log('Email sent: ' + info.response);
      }
    });
});


app.listen(3000, () => {
        console.log('ready server');
});
```
